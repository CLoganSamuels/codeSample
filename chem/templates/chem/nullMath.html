def nullMath(matrix, i):
    coEffs = Matrix(matrix).nullspace()
    #coefficients in fraction form
    notInts = True
    mult = 1
    
    while notInts:
        intCoEffs = []
        for x in range(i):
            if coEffs[0][x] * mult % 1 == 0: #coefficients must be whole numbers
                intCoEffs.append(coEffs[0][x] * mult)
                notInts = False
            else:
                notInts = True
                mult += 1
                break
    return(intCoEffs)
