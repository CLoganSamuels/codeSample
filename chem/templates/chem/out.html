def out(coEffs, reactants, products, moleWeights):
    coEffDict = {}
    equation = ""
    i = 0
    subscript = str.maketrans("0123456789", "₀₁₂₃₄₅₆₇₈₉")
    for reactant in reactants:
        coEffDict[reactant] = coEffs[i]
        if coEffs[i] == 1: coEffs[i] = ''
        equation = equation + str(coEffs[i]) + reactant.translate(subscript) + ' + '
        i += 1
    equation = equation[:-2] + "→ "
    for product in products:
        coEffDict[product] = coEffs[i]
        if coEffs[i] == 1: coEffs[i] = ''
        equation = equation + str(coEffs[i]) + product.translate(subscript) + ' + '
        i += 1
    return((equation[:-2], moleWeights, coEffDict))
